Custom Buttons

You can change the Prev and Next arrows, as you like, as shown in the following example

Slider 2

1. First create the functions that will wrap the new buttons​

exampleButtons.jsx
function CustomNextButton() {
   return (
        <div className="next-icon"></div>
   )
}

function CustomPrevButton() {
    return (
        <div className="prev-icon"></div>
    )
 }

2. Then pass the functions to the slider using props​

exampleButtons.jsx
export default function ExampleButtons() {
    return (
        <Helios
            nextArrow={{ component: <CustomNextButton /> }}
            prevArrow={{ component: <CustomPrevButton /> }}
        >
            <div className="box">
                <h1>
                    Slider 1
                </h1>
            </div>
            <div className="box">
                <h1>
                    Slider 2
                </h1>
            </div>
            {/** More Elements*/}
        </Helios>
    )
}
